    pass

Generator

def count(n):
    for i in range(n):
        yield i

for x in count(3):
    print(x)

Decorator

def deco(func):
    def wrap():
        print("Before")
        func()
        print("After")
    return wrap

@deco
def hello():
    print("Hello")

hello()

Multithreading

import threading

def task():
    print("Running")

t = threading.Thread(target=task)
